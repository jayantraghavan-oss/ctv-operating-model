SELECT ROUND(SUM(gross_spend_usd) / 7, 2) AS avg_daily_gas
FROM `moloco-ae-view.athena.fact_dsp_core`
WHERE date_utc BETWEEN '{D7_START}' AND '{YESTERDAY}'
  AND campaign.os = 'CTV'
  AND moloco_product = 'DSP'

Output avg_daily_gas → gas_per_day
Derived ARR = gas_per_day × 365 · pct = ARR / $100M

Q4 · Top Advertiser Revenue Concentration (Last 30 Days)
Powers: Revenue Concentration donut chart · concentration risk card
Chart · Risk 
WITH gas AS (
  SELECT campaign.tracking_entity AS te,
    SUM(gross_spend_usd) AS total_gas
  FROM `moloco-ae-view.athena.fact_dsp_core`
  WHERE date_utc BETWEEN '{D30_START}' AND '{YESTERDAY}'
    AND campaign.os = 'CTV'
    AND moloco_product = 'DSP'
  GROUP BY 1
),
grand AS (SELECT SUM(total_gas) AS total FROM gas),
ranked AS (
  SELECT te, total_gas,
    ROW_NUMBER() OVER (ORDER BY total_gas DESC) AS rn
  FROM gas
)
SELECT
  r.te,
  ROUND(100.0 * r.total_gas / g.total, 2) AS pct
FROM ranked r CROSS JOIN grand g
WHERE r.rn <= 5
ORDER BY r.rn
Window Rolling 30 days
Grouping campaign.tracking_entity (advertiser)
Output Top 5 pcts + rest = concentration[6]
